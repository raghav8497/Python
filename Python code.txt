stored_pin = "1234"
balance = 10000.0   # Initial balance
max_attempts = 3

print("===== Welcome to ATM Simulation System =====")

# PIN Verification
attempts = 0
while attempts < max_attempts:
    entered_pin = input("Enter your 4-digit PIN: ")

    if entered_pin == stored_pin:
        print("\nPIN Verified Successfully!\n")
        break
    else:
        attempts += 1
        print("Incorrect PIN! Attempts left:", max_attempts - attempts)

if attempts == max_attempts:
    print("\nCard Blocked due to 3 incorrect attempts!")
    exit()

# ATM Menu
while True:
    print("\n====== ATM MENU ======")
    print("1. Balance Enquiry")
    print("2. Deposit Money")
    print("3. Withdraw Money")
    print("4. Exit")

    choice = input("Enter your choice (1-4): ")

    # Balance Enquiry
    if choice == "1":
        print("\nCurrent Balance: Rs.", format(balance, ".2f"))

    # Deposit
    elif choice == "2":
        try:
            amount = float(input("Enter amount to deposit: Rs. "))
            if amount > 0:
                balance += amount
                print("Amount Deposited Successfully!")
                print("Updated Balance: Rs.", format(balance, ".2f"))
            else:
                print("Invalid amount! Enter a positive value.")
        except:
            print("Invalid input! Please enter a number.")

    # Withdraw
    elif choice == "3":
        try:
            amount = float(input("Enter amount to withdraw: Rs. "))
            if amount <= 0:
                print("Invalid amount! Enter a positive value.")
            elif amount > balance:
                print("Insufficient balance!")
            else:
                balance -= amount
                print("Amount Withdrawn Successfully!")
                print("Remaining Balance: Rs.", format(balance, ".2f"))
        except:
            print("Invalid input! Please enter a number.")

    # Exit
    elif choice == "4":
        print("\nThank you for using ATM Simulation System!")
        break

    else:
        print
